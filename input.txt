// SimpleDemo.ets
@Entry
@Component
struct SimpleDemo {
  @State showChild: boolean = true

  aboutToAppear() {
    console.info('SimpleDemo aboutToAppear')
  }

  onDidBuild() {
    console.info('SimpleDemo onDidBuild')
  }

  aboutToDisappear() {
    console.info('SimpleDemo aboutToDisappear')
  }

  build() {
    Column() {
      Button(this.showChild ? '隐藏子组件' : '显示子组件')
        .margin(20)
        .onClick(() => {
          this.showChild = !this.showChild
        })

      if (this.showChild) {
        SimpleChild()
      }
    }
  }
}

@Component
struct SimpleChild {
  aboutToAppear() {
    console.info('SimpleChild aboutToAppear')
  }

  onDidBuild() {
    console.info('SimpleChild onDidBuild')
  }

  aboutToDisappear() {
    console.info('SimpleChild aboutToDisappear')
  }

  build() {
    Text('我是子组件')
      .fontSize(24)
      .margin(10)
  }
}
